select DISTINCT  A.V_POLICY_NO,
A.V_COMPANY_CODE, 
E.V_COMPANY_NAME,
A.V_COMPANY_BRANCH,
A.V_CNTR_STAT_CODE,
(select V_STATUS_DESC from GNMM_POLICY_STATUS_MASTER WHERE V_STATUS_CODE = V_CNTR_STAT_CODE) AS POLICY_STATUS, 
A.D_COMMENCEMENT,
A.D_ISSUE,
A.V_LASTUPD_USER, 
B.V_COMPANY_CODE, 
B.V_CONTACT_NUMBER,
B.V_CONTACT_PERSON,
B.V_CONTACT_PERSON_DESIG,
D.V_ADD_ONE,
D.V_ADD_TWO, 
D.V_ADD_THREE, 
D.V_STATE_CODE, 
D.V_COMPANY_CODE,
D.V_COUNTRY_CODE,
E.V_COMPANY_NAME,
E.V_REGN_NO
from GNMT_POLICY A, GNDT_COMPMOBILE_CONTACTS B, 
GNDT_COMPANY_ADDRESS D , GNMM_COMPANY_MASTER E 
where V_POLICY_NO like 'UG%' AND V_CNTR_STAT_CODE = 'NB010'
AND A.V_COMPANY_CODE=B.V_COMPANY_CODE
AND A.V_COMPANY_CODE=D.V_COMPANY_CODE
AND A.V_COMPANY_CODE=E.V_COMPANY_CODE

ORDER BY 6 DESC
