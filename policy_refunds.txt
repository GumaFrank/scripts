SELECT   V_VOU_NO,TRUNC(D_VOU_DATE) D_VOU_DATE , JHL_GEN_PKG.GET_VOUCHER_POLICY(PV.V_VOU_NO)   POLICY_NUMBER,
  (SELECT V_PLAN_DESC
 FROM GNMT_POLICY POL, GNMM_PLAN_MASTER PROD
 WHERE POL.V_PLAN_CODE = PROD.V_PLAN_CODE
 AND POL.V_POLICY_NO = JHL_GEN_PKG.GET_VOUCHER_POLICY(PV.V_VOU_NO)) V_PLAN_DESC,
  V_VOU_DESC,V_PAYEE_NAME,N_VOU_AMOUNT,
JHL_GEN_PKG.GET_VOUCHER_DATE( V_VOU_NO,'APPROVE') APPROVAL_DATE, JHL_GEN_PKG.GET_VOUCHER_STATUS_USER( V_VOU_NO,'APPROVE') APPROVED_BY
FROM PYMT_VOUCHER_ROOT PM, PYMT_VOU_MASTER PV, GNMM_POLICY_STATUS_MASTER STATUS
WHERE PM.V_MAIN_VOU_NO = PV.V_MAIN_VOU_NO
AND PV.V_VOU_STATUS = STATUS.V_STATUS_CODE
AND N_VOU_AMOUNT > 50000
AND JHL_GEN_PKG.IS_VOUCHER_AUTHORIZED(V_VOU_NO) = 'Y'
 AND TRUNC (D_VOU_DATE) BETWEEN :P_FM_DT AND :P_TO_DT
 AND V_VOU_DESC   LIKE '%DEPOSIT%'
